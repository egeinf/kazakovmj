def f(n):
  if n == 1: return 2
  if n % 2 == 0: return 2*f(n-1)
  if n % 2 != 0: return 2*n+f(n-2)

print(f(33))

def f(n):
  if n < 5: return 5-n
  if n % 3 == 0: return 2 ** (n/3) + (n-5) * f(n-5)
  if n % 3 != 0: return 3*n+3*f(n-1)+f(n-2)

print(f(19))

def f(n):
  if n == 0: return 0
